147  sibr::Vector4f averagePt = sibr::Vector4f(0, 0, 0, 1);

149 
150  const sibr::Mesh::Vertices& vertices = mesh.vertices();
151  //const sibr::Mesh::Normals& normals = mesh.normals();

154  sibr::Vector4f v;
155

158  v[1] = vert[i].y = vertices[i][1];
159  v[2] = vert[i].z = vertices[i][2];

167  maxlen = d;

338  sibr::Vector3f Raycaster::smoothNormal(const sibr::Mesh& mesh, const RayHit& hit)
339  {
340  if (!mesh.hasNormals()) {
341  SIBR_ERR << " cannot compute smoothed normals if the mesh does not have normals " << std::endl;
342  }
343  const sibr::Mesh::Normals& normals = mesh.normals();
344  const sibr::Vector3u& tri = mesh.triangles()[hit.primitive().triID];
345 
346  const float ucoord = hit.barycentricCoord().u;
347  const float vcoord = hit.barycentricCoord().v;
348  float wcoord = 1.f - ucoord - vcoord;
349  wcoord = (wcoord >= 0.0f ? (wcoord <= 1.0f ? wcoord : 1.0f) : 0.0f);
350 
351  return (wcoord * normals[tri[0]] + ucoord * normals[tri[1]] + vcoord * normals[tri[2]]).normalized();
352  }
