34  '''Determine the image type of fhandle and return its size.
35  from draco'''
36  with open(fname, 'rb') as fhandle:
37  head = fhandle.read(24)
38  if len(head) != 24:
39  return
40  if imghdr.what(fname) == 'png':
41  check = struct.unpack('>i', head[4:8])[0]
42  if check != 0x0d0a1a0a:
43  return
44  width, height = struct.unpack('>ii', head[16:24])
45  elif imghdr.what(fname) == 'gif':
46  width, height = struct.unpack('<HH', head[6:10])
47  elif imghdr.what(fname) == 'jpeg':
