10  {
11  float ratio = cam.aspect();
12  float angle = cam.fovy();
13  float nearD = cam.znear();
14  float farD = cam.zfar();
15 
16  // compute width and height of the near and far plane sections
17  float tang = (float)tan(SIBR_DEGTORAD(angle) * 0.5);
18  float nh = nearD * tang;
19  float nw = nh * ratio;
20  float fh = farD * tang;
21  float fw = fh * ratio;
22 
23  Vector3f nc, fc, X, Y, Z;
24  const Vector3f& p = cam.position();
25 
26  // compute the Z axis of camera
27  // this axis points in the opposite direction from
28  // the looking direction
29  Z = -cam.dir();
30 
31  // X axis of camera with given "up" vector and Z axis
32  X = cam.up().cross(Z);
33  X.normalize();
34 
35  // the real "up" vector is the cross product of Z and X
36  Y = cam.up();
37 
38  // compute the centers of the near and far planes
39  nc = p - Z * nearD;
40  fc = p - Z * farD;
41 
42  _planes[NEARP].buildFrom( -Z, nc );
43  _planes[FARP].buildFrom( Z, fc );
44

46

48  aux.normalize();

50  _planes[TOP].buildFrom( normal, nc + Y*nh );
51

53  aux.normalize();
