775  void Image<T_Type, T_NumComp>::fromOpenCV(const cv::Mat& imgSrc) {
776  cv::Mat img = imgSrc.clone();
777 
778  if (img.depth() != opencv::imageType<T_Type>())
779  {
780  img.convertTo(img, opencv::imageType<T_Type>(),
781  opencv::imageTypeRange<T_Type>() / opencv::imageTypeCVRange(img.depth()));
782  }
783 
784  cv::Vec<T_Type, T_NumComp> p;
785  if (img.channels() != T_NumComp)
786  {
787  _pixels = cv::Mat(img.rows, img.cols, opencvType());
788  for (int y = 0; y < img.rows; ++y)
789  {
