62 void irls(Eigen::MatrixX4f mX, Eigen::VectorXf vY, Eigen::Vector4f& mCoeffs, float tune) {
63 
64  Eigen::MatrixXf mX_ = mX;
65  Eigen::VectorXf vY_ = vY;
66  // Find the least squares coefficients.
67  Eigen::Vector4f vC = mX_.jacobiSvd(Eigen::ComputeThinU | Eigen::ComputeThinV).solve(vY);
68  //Log(EInfo, "Finished solving for LS solution.");
69 
70  // Form the leverage value matrix as H = X . ( X_T . X ) . X_T
71  // Form the leverage factor matrix as 1/sqrt(1 - diag(H))
72  Eigen::VectorXf mH = (mX_ * (((mX_.transpose() * mX_).inverse()) * mX_.transpose())).diagonal();
73  Eigen::MatrixXf mH_ = (Eigen::VectorXf::Constant(mH.rows(), 1, 1) - mH).cwiseSqrt().cwiseInverse().asDiagonal();
74 
75  std::cout << vC << std::endl;
76  for (int i = 0; ; i++) {
77  std::cout << "IRLS: Iteration " << i << ":";
78 
79  // Find residuals:
80  Eigen::VectorXf resid = vY - mX * vC;
81 
82  float mad = findMAD(resid.cwiseAbs());
83 
84  // Calcualte Standardized Adjusted Residuals.
85  Eigen::VectorXf r = (mH_ * resid * 0.6745) / (mad * tune);
86 
87  // Find the root weight of residuals.
88  Eigen::VectorXf wt = weight(r);
89 
90  // Multiply X and Y with the root of the weights.
91  mX_ = wt.asDiagonal() * mX;
92  vY_ = wt.asDiagonal() * vY;
93 
94  std::cout << "MAD= " << mad << ", ";
95 
96  // Regress the weighted X and Y to find weighted least squares optimisation.
97  Eigen::Vector4f vC_ = mX_.jacobiSvd(Eigen::ComputeThinU | Eigen::ComputeThinV).solve(vY_);
98 
99  // Find mean deviation in coefficients.
100  float meanDiff = (vC - vC_).cwiseAbs().mean();
101  std::cout << "MD=" << meanDiff << ",";
102  // Terminate if the deviation is too small or number of iterations has been exceeded.
103  if (meanDiff < 0.01f || i > MAX_ITERS) {
104  mCoeffs = vC_;
105  std::cout << "\n";
106  break;
107  }
108 
109  vC = vC_;
110  std::cout << "\n";
111  }

749 void computeRT(std::vector<sibr::Vector3f> A, std::vector<sibr::Vector3f>B, Matrix3f S, Matrix3f& R, sibr::Vector3f& T) {
750 
751  //finding R,T such that B = RA + T;
752  const int numPoint = B.size();
753  //Scaling the points to align
754  std::vector<sibr::Vector3f> AScaled;
755  for (int i = 0; i < numPoint; i++) {
756  AScaled.push_back(S * A[i]);
757  }
758 
759  //Computing the centroids
760  sibr::Vector3f centroidB(0, 0, 0);
761  sibr::Vector3f centroidA(0, 0, 0);
762  for (int i = 0; i < numPoint; i++) {
763  centroidB += B[i];
764  centroidA += AScaled[i];
765  }
766  centroidB /= numPoint;
767  centroidA /= numPoint;
768 
769 
770  //Now we estimate the rotation :
771  Matrix3f H;
772  H << 0, 0, 0,
773  0, 0, 0,
774  0, 0, 0;
775 
776  for (int i = 0; i < numPoint; i++) {
777  H += (AScaled[i] - centroidA) * (B[i] - centroidB).transpose();
778  }
779 
780  Eigen::JacobiSVD<Eigen::MatrixXf> svd(H, Eigen::ComputeThinU | Eigen::ComputeThinV);
781 
782  R = svd.matrixV() * svd.matrixU().transpose();
783  if (R.determinant() < 0) {
784  //R.col(2) *= -1;
785  //std::cout << "Warning : determinant of rotation matrix is negative, multiplying last column by -1" << std::endl;
786  }
787 
788  //Translation estimation :
789  T = -R * centroidA + centroidB;
790 
791 }

793 float computeS(std::vector<sibr::Vector3f> A, std::vector<sibr::Vector3f>B, float& minScale, float& maxScale) {
794  //findin S such that A*S has the same scale as B
795  sibr::Vector3f meanPosA(0, 0, 0);
796  sibr::Vector3f meanPosB(0, 0, 0);
797 
798  for (int i = 0; i < A.size(); i++) {
799  meanPosA += A[i];
800  meanPosB += B[i];
801  }
802 
803  meanPosA /= A.size();
804  meanPosB /= A.size();
805 
806  float scale = 0;
807  for (int i = 0; i < A.size(); i++) {
808 
809  float scale_i = (B[i] - meanPosB).norm() / (A[i] - meanPosA).norm();
810  scale += scale_i;
811 
812  if (scale_i > maxScale)
813  maxScale = scale_i;
814  if (scale_i < minScale)
815  minScale = scale_i;
816  }
817 
818  scale /= A.size();
819 
820  return scale;
821 }
