13  GLuint GLShader::compileShader(const char* shader_code, GLuint type)
14  {
15  std::string shader_type;
16  switch (type) {
17  case GL_VERTEX_SHADER: shader_type = "vertex"; break;
18  case GL_FRAGMENT_SHADER: shader_type = "fragment"; break;
19  case GL_GEOMETRY_SHADER: shader_type = "geometry"; break;
20  default: std::runtime_error("Shader types other than vertex/fragment/geometry not supported");
21  }
22 
23  GLuint id = glCreateShader(type);
24  glShaderSource(id,1,&shader_code,NULL);
25  glCompileShader(id);
26 
27  GLint compiled;
28  glGetShaderiv(id,GL_COMPILE_STATUS, &compiled);
29  if (!compiled) {
30  GLint maxLength;

32  char* infoLog = new char[maxLength+1];

35  SIBR_WRG << "GLSL " << shader_type.c_str() << " shader compilation failed for program "
36  << m_Name.c_str() << std::endl
37  << infoLog << std::endl;
38  delete [] infoLog;
39  (void)glGetError();
40  return 0;
41  }
42  return id;
43  }
