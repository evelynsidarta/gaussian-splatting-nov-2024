190  template <typename T>
191  static Eigen::Quaternion<T> slerp( const Eigen::Quaternion<T>& q1, const Eigen::Quaternion<T>& q2, float t ) {
192  Eigen::Quaternion<T> q3;
193  float dot = q1.dot(q2);// Eigen::Quaternion<T>::dot(q1, q2);
194  // dot = cos(theta)
195  // if (dot < 0), q1 and q2 are more than 90 degrees apart,
196  // so we can invert one to reduce spinning
197  if (dot < 0)
198  {
199  dot = -dot;
200  q3 = -q2;
201  } else q3 = q2;
202  if (dot < 0.95f)
203  {
204  float angle = acosf(dot);
205  return (q1*sinf(angle*(1-t)) + q3*sinf(angle*t))/sinf(angle);
206  } else // if the angle is small, use linear interpolation
207  return lerp(q1,q3,t);
208  }
