128  void MRFSolver::buildGraphAlphaExp(int label_iteration_id)
129  {
130  double infty = 1 << 25;
131  int num_nodes = (int)_neighborMap->size();
132  int n_nodes_estimation = num_nodes;
133  int n_edges_estimation = num_nodes * 4;
134 
135  _graph = new GraphType(n_nodes_estimation, n_edges_estimation);
136 
137  //add nodes associated to pixels
138  int node_id = 0;
139  for (int p = 0; p < num_nodes; p++) {
140  _graph->add_node();
141 
142  if (_labels[p] == label_iteration_id) {
143  _graph->add_tweights(node_id, unaryTotal(p, label_iteration_id), infty);
144  } else {
145  _graph->add_tweights(node_id, unaryTotal(p, label_iteration_id), unaryTotal(p, _labels[p]));
146  }
147  ++node_id;
148  }
149 
150  //add nodes associated to connexions between pixels
151  for (int p = 0; p < num_nodes; p++) {
152 
153  std::vector<int> & neighors = (*_neighborMap)[p];
154  for (int q_id = 0; q_id < (int)neighors.size(); q_id++) {
155 
156  int q = neighors[q_id];
157 
158  if (p == q) { std::cerr << "!"; }
159  if (q < p) { continue; }
160 
161  if (_labels[p] != _labels[q]) {
162  //extra node associated to edge {p,q}
163  _graph->add_node();
164 
165  _graph->add_tweights(node_id, 0, pairwiseTotal(q, p, _labels[q], _labels[p]));
166
