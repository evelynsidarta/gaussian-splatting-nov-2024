137  void Orbit::interpolateOrbit() {
138  using namespace Eigen;
139 
140  float k = (_orbit.factor) / (float)(SIBR_ORBIT_INTERPOLATE_FRAMES);
141  bool keepCam = _orbit.keepCamera;
142 
143  float theta = (_orbit.status == OrbitParameters::FORWARD_X ? (float)(SIBR_2PI * k) : _orbit.theta);
144  float phi = (_orbit.status == OrbitParameters::FORWARD_Y ? (float)(SIBR_2PI * k) : _orbit.phi);
145  float roll = (_orbit.status == OrbitParameters::FORWARD_Z ? (float)(SIBR_2PI * k) : _orbit.roll);
146 
147  sibr::Vector3f dir = -(_orbit.zAxis);
148  sibr::Quaternionf qRoll(AngleAxisf(roll, _orbit.zAxis));
149  sibr::Quaternionf qTheta(AngleAxisf(theta, _orbit.yAxis));
150  sibr::Quaternionf qPhi(AngleAxisf(phi, _orbit.xAxis));
151 
152  sibr::Vector3f center = _orbit.center;
153  sibr::Vector3f Eye = center + _orbit.radius*((qTheta*qPhi)*(dir));
154  sibr::Vector3f up(qRoll*_orbit.yAxis);
155 
156  sibr::Camera n(_orbit.initialCamera);
157  n.setLookAt(Eye, center, up);
158  n.aspect(_orbit.initialCamera.aspect());
159 
160 
161  if (_orbit.status == OrbitParameters::STATIC || keepCam) {
162  sibr::Quaternionf qTot = qTheta*qPhi*qRoll;
163  _orbit.xAxis = qTot*_orbit.xAxis;
164  _orbit.yAxis = qTot*_orbit.yAxis;
165  _orbit.zAxis = qTot*_orbit.zAxis;
166 
167  _orbit.theta = 0;
168  _orbit.phi = 0;
169  _orbit.roll = 0;
170  }
171  else {
172  _orbit.factor += _orbit.direction;
173  }
174 
175  if (keepCam) {
176  _orbit.status = OrbitParameters::STATIC;
177  _orbit.keepCamera = false;
178  }
179 
180  _currentCamera = sibr::InputCamera(n, _currentCamera.w(), _currentCamera.h());
