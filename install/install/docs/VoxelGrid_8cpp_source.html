160  std::vector<size_t> VoxelGridBase::rayMarch(const Ray & ray) const
161  {
162  sibr::Vector3f start = ray.orig();
163 
164  if (!isInside(start)) {
165  sibr::Vector3f intersection;
166  if (intersectionWithBox(ray, intersection)) {
167  start = intersection;
168  } else {
169  return {};
170  }
171  }
172 
173  start = start.cwiseMax(box.min()).cwiseMin(box.max() - 0.01f*getCellSize());
174 
175  sibr::Vector3i currentVoxel = getCell(start);
176 
177  sibr::Vector3i steps = ray.dir().unaryExpr([](float f) { return f >= 0 ? 1 : -1; }).cast<int>();
178 
179  const sibr::Vector3f deltas = getCellSize().cwiseQuotient(ray.dir().cwiseAbs());
180  const sibr::Vector3f frac = (start - box.min()).cwiseQuotient(getCellSize()).unaryExpr([](float f) { return f - std::floor(f); });
181  sibr::Vector3i finalVoxels;
182  sibr::Vector3f ts;
183  for (int c = 0; c < 3; c++) {
184  ts[c] = deltas[c] * (ray.dir()[c] >= 0 ? 1.0f - frac[c] : frac[c]);
185  finalVoxels[c] = (ray.dir()[c] >= 0 ? dims[c] : -1);
186  }
187 
188  std::vector<size_t> visitedCellsIds;
189  while (true) {
190  visitedCellsIds.push_back(getCellId(currentVoxel));
191 
192  int c = getMinIndex(ts);
193  currentVoxel[c] += steps[c];
194  if (currentVoxel[c] == finalVoxels[c]) {
195  break;
196  }
197  ts[c] += deltas[c];
198  }
