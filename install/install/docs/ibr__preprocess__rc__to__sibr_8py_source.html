44 def get_image_size(fname):
45  '''Determine the image type of fhandle and return its size.
46  from draco'''
47  with open(fname, 'rb') as fhandle:
48  head = fhandle.read(24)
49  if len(head) != 24:
50  return
51  if imghdr.what(fname) == 'png':
52  check = struct.unpack('>i', head[4:8])[0]
53  if check != 0x0d0a1a0a:
54  return
55  width, height = struct.unpack('>ii', head[16:24])
56  elif imghdr.what(fname) == 'gif':
57  width, height = struct.unpack('<HH', head[6:10])
58  elif imghdr.what(fname) == 'jpeg':
59  try:
60  fhandle.seek(0) # Read 0xff next
61  size = 2
62  ftype = 0
63  while not 0xc0 <= ftype <= 0xcf:
64  fhandle.seek(size, 1)
65  byte = fhandle.read(1)
66  while ord(byte) == 0xff:
67  byte = fhandle.read(1)
68  ftype = ord(byte)
69  size = struct.unpack('>H', fhandle.read(2))[0] - 2
70  # We are at a SOFn block
71  fhandle.seek(1, 1) # Skip `precision' byte.
72  height, width = struct.unpack('>HH', fhandle.read(4))
73  except Exception: # IGNORE:W0703
74  return
75  else:
76  return
77  return width, height

129 def get_scale_factor (current_res, target_res):
130  """Return the scale factor needed to go from current_res to target_res.
131  The value is calculated based on the idea that we don't want to crop the input dataset anymore
132  and that we prefer to add black borders in order to reach the target resolution
133 
134  Args:
135  current_res (vec2): current dataset resolution
136  target_res (vec2): target dataset resolution
137 
138  Returns:
139  float: scale factor
140  """
141 
142  # to go from current to target resolution, we have to options:
143  # either scale down current width to match target width and modify height accordingly
144  # or scalen down current height to match target height and modify width accordingly.
145  # we take the option that doesn't crop the remaining dimension and add the least black border
146 
147  # trying scaling down width
148  alpha_by_width = (float)(target_res[0]) / current_res[0]
149  adjusted_height = (int) (alpha_by_width * current_res[1])
150  delta_height = target_res[1] - adjusted_height
151 
152  # trying scaling down height
153  alpha_by_height = (float)(target_res[1]) / current_res[1]
154  adjusted_width = (int) (alpha_by_height * current_res[0])
155  delta_width = target_res[0] - adjusted_width
156 
157  # since we don't want to crop the input images even more, we considerer only
158  # the options were black borders need to be added
159 
160  if (delta_height < 0):
161  # height would need to be cropped. take the other option
162  return alpha_by_height
163  elif (delta_width < 0):
164  # width would need to be cropped. take the other option
165  return alpha_by_width
166  else:
167  # none of them need to be cropped. take option that adds less black border
168  return alpha_by_width if (delta_height < delta_width) else alpha_by_height
