360  inline sibr::Mesh::Ptr VoxelGrid<CellType>::getAllCellMeshWithIds(bool filled, std::vector<std::size_t> cell_ids) const
361  {
362  int numNonZero = (int)cell_ids.size();
363 
364  auto out = std::make_shared<sibr::Mesh>();
365 
366  sibr::Mesh::Ptr baseMesh = filled ? baseCellMeshFilled : baseCellMesh;
367 
368  const int numT = (int)baseMesh->triangles().size();
369  const int numTtotal = numNonZero * numT;
370  const int numV = (int)baseMesh->vertices().size();
371  const int numVtotal = numNonZero * numV;
372  const sibr::Vector3u offsetT = sibr::Vector3u(numV, numV, numV);
373 
374  sibr::Mesh::Vertices vs(numVtotal);
375  sibr::Mesh::Triangles ts(numTtotal);
376  for (int i = 0; i < numNonZero; ++i) {
377  const auto cell = getCell(cell_ids[i]);
378  const sibr::Vector3f offsetV = cell.cast<float>().array() * getCellSize().array();
379 
380  for (int v = 0; v < numV; ++v) {
381  vs[i * numV + v] = baseMesh->vertices()[v] + offsetV;
382  }
383  for (int t = 0; t < numT; ++t) {
384  ts[i * numT + t] = baseMesh->triangles()[t] + i * offsetT;
385  }
386  }
387 
388  out->vertices(vs);
389  out->triangles(ts);
390  return out;
391  }
