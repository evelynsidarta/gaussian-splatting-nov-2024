143 void sibr::ULRV3View::updateCameras(bool allowResetToDefault) {
144  // If we are here, the rendering mode or the selected index have changed, we need to update the enabled cameras.
145  std::vector<uint> imgs_ulr;
146  const auto & cams = _scene->cameras()->inputCameras();
147 
148  // Compute the cameras indices based on the new mode.
149  if (_renderMode == RenderMode::ONE_CAM) {
150  // We only use the given camera (if it is active).
151  if (cams[_singleCamId]->isActive()) {
152  imgs_ulr = { (uint)_singleCamId };
153  } else {
154  std::cerr << "The camera is not active, using all cameras." << std::endl;
155  }
156  } else if (_renderMode == RenderMode::LEAVE_ONE_OUT) {
157  // We use all active cameras apart from the one given.
158  for (size_t cid = 0; cid < cams.size(); ++cid) {
159  if (cid != (size_t)_singleCamId && cams[cid]->isActive()) {
160  imgs_ulr.push_back(uint(cid));
161  }
162  }
163  }
164  else if (_renderMode == RenderMode::EVERY_N_CAM) {
165  // We pick one camera every N
166  for (size_t cid = 0; cid < cams.size(); ++cid) {
167  if ((cid % _everyNCamStep == 0) && cams[cid]->isActive()) {
168  imgs_ulr.push_back(uint(cid));
169  }
170  }
171  } else if(allowResetToDefault){
172  // We use all active cameras.
173  for (size_t cid = 0; cid < cams.size(); ++cid) {
174  if (cams[cid]->isActive()) {
175  imgs_ulr.push_back(uint(cid));
176  }
177  }
178  }
179  // Only update if there is at least one camera enabled.
180  if(!imgs_ulr.empty()) {
181  // Update the shader informations in the renderer.
182  _ulrRenderer->updateCameras(imgs_ulr);
183  // Tell the scene which cameras we are using for debug visualization.
184  _scene->cameras()->debugFlagCameraAsUsed(imgs_ulr);
185  }
