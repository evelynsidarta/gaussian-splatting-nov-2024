334  void MultiViewBase::mosaicLayout(const Viewport & vp)
335  {
336  const int viewsCount = numSubViews();
337 
338  // Do square decomposition for now.
339  // Find the next square.
340  const int sideCount = int(std::ceil(std::sqrt(viewsCount)));
341  const int verticalShift = int(ImGui::GetTitleBarHeight());
342 
343  Viewport usedVP = Viewport(vp.finalLeft(), vp.finalTop() + verticalShift, vp.finalRight(), vp.finalBottom());
344  Vector2f itemRatio = Vector2f(1, 1) / sideCount;
345 
346  int vid = 0;
347  for (auto & view : _ibrSubViews) {
348  // Compute position on grid.
349  const int col = vid % sideCount;
350  const int row = vid / sideCount;
351  view.second.viewport = Viewport(usedVP, col*itemRatio[0], row * itemRatio[1], (col + 1)*itemRatio[0], (row + 1)*itemRatio[1]);
352  view.second.shouldUpdateLayout = true;
353  ++vid;
354  }
355  for (auto & view : _subViews) {
356  // Compute position on grid.
357  const int col = vid % sideCount;
358  const int row = vid / sideCount;
359  view.second.viewport = Viewport(usedVP, col*itemRatio[0], row * itemRatio[1], (col + 1)*itemRatio[0], (row + 1)*itemRatio[1]);
360  view.second.shouldUpdateLayout = true;
361  ++vid;
362  }
363  for (auto & view : _subMultiViews) {
364  // Compute position on grid.
365  const int col = vid % sideCount;
366  const int row = vid / sideCount;
367  view.second->mosaicLayout(Viewport(usedVP, col*itemRatio[0], row * itemRatio[1], (col + 1)*itemRatio[0], (row + 1)*itemRatio[1]));
368  ++vid;
369  }
