35 static void rasterizeLine(uint8_t *dest, int destWidth, const int *p1, const int *p2, const sibr::Vector3ub & color)
36 {
37  const int dx = abs(p2[0] - p1[0]), sx = p1[0] < p2[0] ? 1 : -1;
38  const int dy = abs(p2[1] - p1[1]), sy = p1[1] < p2[1] ? 1 : -1;
39  int err = (dx > dy ? dx : -dy) / 2;
40  int current[2];
41  current[0] = p1[0];
42  current[1] = p1[1];
43  while(setPixel(dest, destWidth, current[0], current[1], color), current[0] != p2[0] || current[1] != p2[1]){
44  const int e2 = err;
45  if (e2 > -dx) { err -= dy; current[0] += sx; }
46  if (e2 < dy) { err += dx; current[1] += sy; }
47  }
48 }

64 void rasterizeTriangle(uint8_t *dest, int destWidth, const int *t0, const int *t1, const int *t2, const sibr::Vector3ub & color)
65 {
66  if (t0[1] > t1[1]) std::swap(t0, t1);
67  if (t0[1] > t2[1]) std::swap(t0, t2);
68  if (t1[1] > t2[1]) std::swap(t1, t2);
69  const int total_height = t2[1] - t0[1];
70  for (int i = 0; i < total_height; i++) {
71  const bool second_half = i > t1[1] - t0[1] || t1[1] == t0[1];
72  const int segment_height = second_half ? t2[1] - t1[1] : t1[1] - t0[1];
73  const float alpha = (float)i / total_height;
74  const float beta = (float)(i - (second_half ? t1[1] - t0[1] : 0)) / float(segment_height);
75  int A[2], B[2];
76  for (int j = 0; j < 2; j++) {
77  A[j] = int(t0[j] + (t2[j] - t0[j]) * alpha);
78  B[j] = int(second_half ? t1[j] + (t2[j] - t1[j]) * beta : t0[j] + (t1[j] - t0[j]) * beta);
79  }
80  if (A[0] > B[0]) std::swap(A, B);
81  for (int j = A[0]; j <= B[0]; j++) {
82  setPixel(dest, destWidth, j, t0[1] + i, color);
83  }
84  }
85 }
