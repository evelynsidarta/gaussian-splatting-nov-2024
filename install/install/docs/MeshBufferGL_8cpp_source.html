89  {
90  // Create buffer for indices (called elements in opengl)
91  std::vector<GLuint> indices;
92 
93  if(adjacency) {
94 
95  std::unordered_map<std::pair<GLuint, GLuint>, GLuint, hash_pair> trianglesByEdges(mesh.triangles().size() * 3);

97 
98  for(int i = 0; i < mesh.triangles().size(); i++)
99  {
100  // store triangles vertices by edges
101 
102  trianglesByEdges.insert({ { mesh.triangles()[i][0], mesh.triangles()[i][1] }, mesh.triangles()[i][2] });
103  trianglesByEdges.insert({ { mesh.triangles()[i][1], mesh.triangles()[i][2] }, mesh.triangles()[i][0] });
104  trianglesByEdges.insert({ { mesh.triangles()[i][2], mesh.triangles()[i][0] }, mesh.triangles()[i][1] });
105  }
106 
107  for (int i = 0; i < mesh.triangles().size(); i++)
108  {
109  // input triangle
110  // 1 - 2
111  // \ /
112  // 0
113 
114  // adjacency list
115  // 3
116  // / \
117  // 2 - 4
118  // / \ / \
119  // 1 - 0 - 5
120 
121  // use reverse edges to find adjacent triangles
122 
123  // 0
124  indices.push_back(mesh.triangles()[i][0]);
125  // 1
126  indices.push_back(trianglesByEdges[{ mesh.triangles()[i][1], mesh.triangles()[i][0] }]);
127  // 2

129  // 3

131  // 4
132  indices.push_back(mesh.triangles()[i][2]);
133  // 5

135  }

155  void MeshBufferGL::build( const Mesh& mesh, bool adjacency )
156  {
157  if (!_vaoId)
158  {
159  glGenVertexArrays(1, &_vaoId);
160  glGenBuffers(BUFCOUNT, &_bufferIds[0]);
161  }
162 
163  glBindVertexArray(_vaoId);
164 
165  CHECK_GL_ERROR;
166 
167  fetchIndices(mesh, false);
168 
169  if(adjacency)
170  fetchIndices(mesh, true);
171 
172  uint numVertices = (uint)mesh.vertices().size();
173  _vertexCount = numVertices;
174  //SIBR_DEBUG(mesh.triangles().size());
175  std::vector<GLfloat> vertices = prepareVertexData<GLfloat>(
176  mesh.vertices(), numVertices);
177  std::vector<GLfloat> colors = prepareVertexData<GLfloat>(
178  mesh.colors(), numVertices);
179  std::vector<GLfloat> texcoords = prepareVertexData<GLfloat>(
180  mesh.texCoords(), numVertices);
181  std::vector<GLfloat> normals = prepareVertexData<GLfloat>(
182  mesh.normals(), numVertices);
183 
184  // Following this order:
185  //VertexAttribLocation = 0,
186  //ColorAttribLocation = 1,
187  //TexCoordAttribLocation = 2,
188  //NormalAttribLocation = 3,
189 
190  // Every data (from different types) are all put together into vertexData
191  std::vector<uint8> vertexData;
192  vertexData.reserve(
193  getVectorDataSize(vertices) +
194  getVectorDataSize(colors) +
195  getVectorDataSize(texcoords)+
196  getVectorDataSize(normals));
197 
198  appendVertexData(vertexData, vertices);
199  appendVertexData(vertexData, colors);
200  appendVertexData(vertexData, texcoords);
201  appendVertexData(vertexData, normals);
202 
203  glBindBuffer(GL_ARRAY_BUFFER, _bufferIds[BUFVERTEX]);
