212  {

231  bool MeshTexturing::sampleNeighborhood(int px, int py, RayHit & hit)
232  {
233  bool hasHit = hitTest(px, py, hit);
234  // Sample a 3x3 neighborhood to counter-act aliasing/interpolation later on, as long as we don't get a hit.
235  // The order is important, to first fetch in line/column and then in diagonal. Sorry for the cache...
236  std::vector<int> dxs = { px, px - 1, px + 1 };
237  std::vector<int> dys = { py, py - 1, py + 1 };
238  for (const int dx : dxs) {
239  if (hasHit) {
240  break;
241  }
242  for (const int dy : dys) {
243  if (hasHit) {
244  break;
245  }
246  // Ignore center pixel, already tested.
247  if (dx == px && dy == py) {
248  continue;
249  }
250  hasHit = hitTest(dx, dy, hit);
251  }
252  }
253  return hasHit;
254  }
