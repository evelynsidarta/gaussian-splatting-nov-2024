83  void DatasetView::repro(ReprojectionData & data)
84  {
85  const Vector3f & pt = data.point3D;
86  for (int im = 0; im<(int)cams.size(); ++im) {
87  const auto & cam = cams[im];
88  if (!cam.frustumTest(pt)) {
89  continue;
90  }
91  Vector3f pt2d = cam.projectImgSpaceInvertY(pt);
92 
93  if (data.occlusionTest) {
94  float dist = (cam.position() - pt).norm();
95  Ray ray = Ray(pt, (cam.position() - pt).normalized());
96  auto hit = proxyData().raycaster->intersect(ray, 0.01f);
97  if (hit.hitSomething() && std::abs(hit.dist() - dist) / dist > 0.01f) {
98  continue;
