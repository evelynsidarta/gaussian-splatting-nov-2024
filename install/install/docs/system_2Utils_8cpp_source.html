166  {
167  boost::filesystem::path source = src;
168  boost::filesystem::path destination = dst;
169  namespace fs = boost::filesystem;
170  try
171  {
172  // Check whether the function call is valid
173  if (!fs::exists(source) || !fs::is_directory(source))
174  {
175  std::cerr << "Source directory " << source.string()
176  << " does not exist or is not a directory." << '\n'
177  ;
178  return false;
179  }
180  if (fs::exists(destination))
181  {
182  std::cerr << "Destination directory " << destination.string()
183  << " already exists." << '\n'
184  ;
185  return false;
186  }
187  // Create the destination directory
188  if (!fs::create_directory(destination))
189  {
190  std::cerr << "Unable to create destination directory"
191  << destination.string() << '\n'
192  ;
193  return false;
194  }
195  }
196  catch (fs::filesystem_error const & e)
197  {
198  std::cerr << e.what() << '\n';
199  return false;
200  }
201  // Iterate through the source directory
202  for (fs::directory_iterator file(source); file != fs::directory_iterator(); ++file)
203  {
204  try
205  {
206  fs::path current(file->path());
207  if (fs::is_directory(current))
208  {
209  // Found directory: Recursion
210  if (!copyDirectory(current.string(), (destination / current.filename()).string()))
211  {
212  return false;
213  }
214  }
215  else
216  {
217  // Found file: Copy
218  fs::copy_file(
219  current,
220  destination / current.filename()
221  );
222  }
223  }
224  catch (fs::filesystem_error const & e)

227  }
