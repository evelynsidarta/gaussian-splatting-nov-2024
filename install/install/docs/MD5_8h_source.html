127 /* F, G and H are basic MD5 functions: selection, majority, parity */
128 #define F(x, y, z) (((x) & (y)) | ((~x) & (z)))
129 #define G(x, y, z) (((x) & (z)) | ((y) & (~z)))
130 #define H(x, y, z) ((x) ^ (y) ^ (z))
131 #define I(x, y, z) ((y) ^ ((x) | (~z)))
132 
133 /* ROTATE_LEFT rotates x left n bits */
134 #define ROTATE_LEFT(x, n) (((x) << (n)) | ((x) >> (32-(n))))
135 
136 /* FF, GG, HH, and II transformations for rounds 1, 2, 3, and 4 */
137 /* Rotation is separate from addition to prevent recomputation */
138 #define FF(a, b, c, d, x, s, ac) \
139  {(a) += F ((b), (c), (d)) + (x) + (UINT4)(ac); \
140  (a) = ROTATE_LEFT ((a), (s)); \
141  (a) += (b); \
142  }
143 #define GG(a, b, c, d, x, s, ac) \
144  {(a) += G ((b), (c), (d)) + (x) + (UINT4)(ac); \
145  (a) = ROTATE_LEFT ((a), (s)); \
146  (a) += (b); \
147  }
148 #define HH(a, b, c, d, x, s, ac) \
149  {(a) += H ((b), (c), (d)) + (x) + (UINT4)(ac); \
150  (a) = ROTATE_LEFT ((a), (s)); \
151  (a) += (b); \
152  }
153 #define II(a, b, c, d, x, s, ac) \
154  {(a) += I ((b), (c), (d)) + (x) + (UINT4)(ac); \
155  (a) = ROTATE_LEFT ((a), (s)); \
156  (a) += (b); \
157  }

171 void MD5Update (MD5_CTX *mdContext, unsigned char *inBuf, size_t inLen)
172 {
173  UINT4 in[16];
174  int mdi;
175  unsigned int i, ii;
176 
177  /* compute number of bytes mod 64 */
178  mdi = (int)((mdContext->i[0] >> 3) & 0x3F);
179 
180  /* update number of bits */
181  if ((mdContext->i[0] + ((UINT4)inLen << 3)) < mdContext->i[0])
182  mdContext->i[1]++;
183  mdContext->i[0] += ((UINT4)inLen << 3);
184  mdContext->i[1] += ((UINT4)inLen >> 29);
185 
186  while (inLen--) {
187  /* add new character to buffer, increment mdi */
188  mdContext->in[mdi++] = *inBuf++;
189 
190  /* transform if necessary */
191  if (mdi == 0x40) {
192  for (i = 0, ii = 0; i < 16; i++, ii += 4)
193  in[i] = (((UINT4)mdContext->in[ii+3]) << 24) |
194  (((UINT4)mdContext->in[ii+2]) << 16) |
195  (((UINT4)mdContext->in[ii+1]) << 8) |
196  ((UINT4)mdContext->in[ii]);
197  Transform (mdContext->buf, in);
198  mdi = 0;
199  }
200  }

203 void MD5Final (MD5_CTX *mdContext)
204 {
205  UINT4 in[16];
206  int mdi;
207  unsigned int i, ii;
208  unsigned int padLen;
209 
210  /* save number of bits */
211  in[14] = mdContext->i[0];
212  in[15] = mdContext->i[1];
213 
214  /* compute number of bytes mod 64 */
215  mdi = (int)((mdContext->i[0] >> 3) & 0x3F);
216 
217  /* pad out to 56 mod 64 */
218  padLen = (mdi < 56) ? (56 - mdi) : (120 - mdi);
219  MD5Update (mdContext, PADDING, padLen);
220 
221  /* append length in bits and transform */
222  for (i = 0, ii = 0; i < 14; i++, ii += 4)
223  in[i] = (((UINT4)mdContext->in[ii+3]) << 24) |
224  (((UINT4)mdContext->in[ii+2]) << 16) |
225  (((UINT4)mdContext->in[ii+1]) << 8) |
226  ((UINT4)mdContext->in[ii]);
227  Transform (mdContext->buf, in);
228 
229  /* store buffer in digest */
230  for (i = 0, ii = 0; i < 4; i++, ii += 4) {
231  mdContext->digest[ii] = (unsigned char)(mdContext->buf[i] & 0xFF);
232  mdContext->digest[ii+1] =
233  (unsigned char)((mdContext->buf[i] >> 8) & 0xFF);
234  mdContext->digest[ii+2] =
235  (unsigned char)((mdContext->buf[i] >> 16) & 0xFF);
236  mdContext->digest[ii+3] =
237  (unsigned char)((mdContext->buf[i] >> 24) & 0xFF);
238  }
239 }
