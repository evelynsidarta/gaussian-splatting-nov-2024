279  void InteractiveCameraHandler::interpolate() {
280  if (_interpPath.empty()) {
281  return;
282  }
283 
284  // If we reach the last frame of the interpolation b/w two cameras, skip to next camera.
285  if (_interpFactor == SIBR_INTERPOLATE_FRAMES - 1)
286  {
287  _interpFactor = 0;
288  _startCam++;
289  }
290 
291  // If we reach the last camera, restart the interpolation.
292  if (_startCam >= _interpPath.size() - 1) {
293  _interpFactor = 0;
294  _startCam = 0;
295  }
296 
297  float k = std::min(std::max(((_interpFactor) / (float)SIBR_INTERPOLATE_FRAMES), 1e-6f), 1.0f - 1e-6f);
298 
299  sibr::InputCamera & camStart = *_interpPath[_startCam];
300  sibr::InputCamera & camNext = *_interpPath[_startCam + 1];
301  const sibr::Camera cam = sibr::Camera::interpolate(camStart, camNext, k);
302  _currentCamera = sibr::InputCamera(cam, camStart.w(), camStart.h());
303  _currentCamera.aspect(_viewport.finalWidth() / _viewport.finalHeight());
304 
305 
306  _interpFactor = _interpFactor + 1;
307  }
