361  void Window::setFullscreen(const bool fullscreen) {
362  const bool currentState = isFullscreen();
363  if((fullscreen && currentState) || (!fullscreen && !currentState)) {
364  // Do nothing.
365  return;
366  }
367  if (fullscreen) {
368  _oldPosition = position();
369  _oldSize = size();
370  GLFWmonitor* monitor = glfwGetPrimaryMonitor();
371  const GLFWvidmode* mode = glfwGetVideoMode(monitor);
372  glfwSetWindowMonitor(_glfwWin.get(), monitor, 0, 0, mode->width, mode->height, mode->refreshRate);
373  // There is a bug in glfw (see https://github.com/glfw/glfw/issues/1072).
374  // We have to manually re-set the swap interval.
375  glfwSwapInterval(_useVSync ? 1 : 0);
376  } else {
377  glfwSetWindowMonitor(_glfwWin.get(), NULL, _oldPosition[0], _oldPosition[1], _oldSize[0], _oldSize[1], 0);
378  }
